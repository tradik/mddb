Quick Start

# Run MDDB server
docker run -d \
  -p 11023:11023 \
  -p 11024:11024 \
  -v mddb-data:/data \
  tradik/mddb:latest

# Test the API
curl http://localhost:11023/health
# Clone repository
git clone https://github.com/tradik/mddb.git
cd mddb

# Start all services (MDDB + Panel + MCP)
make docker-up

# Services available:
# - MDDB HTTP: http://localhost:11023
# - MDDB gRPC: localhost:11024
# - Web Panel: http://localhost:3000
# - MCP Server: http://localhost:9000
# Download binary
wget https://github.com/tradik/mddb/releases/download/v2.0.4/mddbd-linux-amd64
chmod +x mddbd-linux-amd64

# Run server
./mddbd-linux-amd64

# Server starts on http://localhost:11023
# Clone repository
git clone https://github.com/tradik/mddb.git
cd mddb

# Build
make build

# Run
./bin/mddbd

# Option 1: Docker (Easiest - No Build Required) ‚≠ê

# Prerequisites: MDDB server running on localhost:11023 and localhost:11024
# (See Quick Start section above to start MDDB)

# Configure Windsurf (~/.windsurf/mcp.json)
{
  "mcpServers": {
    "mddb": {
      "command": "docker",
      "args": [
        "run", "-i", "--rm", "--network", "host",
        "-e", "MDDB_GRPC_ADDRESS=localhost:11024",
        "-e", "MDDB_REST_BASE_URL=http://localhost:11023",
        "tradik/mddb:mcp"
      ]
    }
  }
}

# Restart Windsurf - Done!
# MCP will automatically pull tradik/mddb:mcp on first use

# ---

# Option 2: Local Binary (For Development)

# 1. Build MCP stdio binary
cd services/mddb-mcp
make build-stdio

# 2. Configure Windsurf (~/.windsurf/mcp.json)
{
  "mcpServers": {
    "mddb": {
      "command": "/full/path/to/mddb-mcp-stdio",
      "env": {
        "MDDB_GRPC_ADDRESS": "localhost:11024",
        "MDDB_REST_BASE_URL": "http://localhost:11023"
      }
    }
  }
}

# 3. Restart Windsurf IDE
curl http://localhost:11023/health

# Response:
{
  "status": "healthy",
  "mode": "wr"
}
